#Generic scripted effects for scripted progress bars

#Stores the value of a specified progress bar from a specified journal_entry in a var
br_store_progress_bar_value_in_var = {
	je:$journal_entry_name$ = {
		set_global_variable = {
			name = temporary_progress_bar_value_var
			value = $progress_bar$
		}
	}
	set_variable = {
		name = $variable_name$
		value = global_var:temporary_progress_bar_value_var
	}
}
