je_tge_golden_eagle = {
	icon = "gfx/interface/icons/event_icons/event_aztec.dds"
	group = je_group_internal_affairs
	
	should_be_pinned_by_default = yes
	
	immediate = {
		set_variable = {
			name = tge_golden_eagle_var
			value = 100
		}
		set_variable = {
			name = tge_golden_eagle_progress_var
			value = 0
		}
	}

	modifiers_while_active = {
		modifier_tge_golden_aspirations
	}

	scripted_progress_bar = tge_golden_eagle_progressbar
	
	complete = {
		custom_tooltip = {
			text = je_tge_golden_eagle_complete
			var:tge_golden_eagle_var >= 500
		}
		has_law = law_type:law_monarchy
		has_law = law_type:law_autocracy
		has_law = law_type:law_state_religion
		country_rank >= rank_value:major_power
	}
	
	on_complete = {
	}
	
	fail = {
		OR = {
			custom_tooltip = {
				text = je_tge_golden_eagle_fail
				var:tge_golden_eagle_var <= 0
			}
			NOT = {
				country_has_state_religion = rel:br_tge_religion_teotl
			}
		}
	}
	
	on_fail = {
	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = tge_golden_eagle_progress_var
				value = tge_golden_eagle_progress_value
			}
			scope:journal_entry = {
				set_bar_progress = {
					name = tge_golden_eagle_progressbar
					value = owner.var:tge_golden_eagle_var
				}
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = tge_golden_eagle_var
				add = var:tge_golden_eagle_progress_var
			}
			clamp_variable = {
				name = tge_golden_eagle_var
				max = 500
				min = 0
			}
			scope:journal_entry = {
				set_bar_progress = {
					name = tge_golden_eagle_progressbar
					value = owner.var:tge_golden_eagle_var
				}
			}
		}
	}
	
	weight = 1000
	
}
