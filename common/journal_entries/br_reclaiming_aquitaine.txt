je_reclaiming_aquitaine = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		country_rank >= rank_value:great_power
		exists = c:HRE
		this = c:HRE
		exists = c:OCC
	}

	possible = {
		NOT = { country_has_primary_culture = cu:frankish }
		exists = c:OCC
		game_date > 1776.1.1
	}
	immediate = {
		s:STATE_GUYENNE = {
			add_claim = c:HRE
		}
		s:STATE_AUVERGNE_LIMOUSIN = {
			add_claim = c:HRE
		}
		s:STATE_AQUITAINE = {
			add_claim = c:HRE
		}
		s:STATE_LANGUEDOC = {
			add_claim = c:HRE
		}

	}

	complete = {
		NOT = { exists = c:OCC }
		owns_entire_state_region = STATE_LANGUEDOC
		owns_entire_state_region = STATE_RHONE
		owns_entire_state_region = STATE_GUYENNE
		owns_entire_state_region = STATE_AUVERGNE_LIMOUSIN
		owns_entire_state_region = STATE_AQUITAINE
	}

	on_complete = {
		add_treasury = 75000
		add_loyalists = {
			pop_type = aristocrats
			value = very_large_radicals
		}
		add_loyalists = {
			pop_type = officers
			value = very_large_radicals
		}
		add_loyalists = {
			pop_type = soldiers
			value = very_large_radicals
		}
		add_loyalists = {
			pop_type = clergymen
			value = very_large_radicals
		}
		add_loyalists = {
			pop_type = bureaucrats
			value = very_large_radicals
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
}
