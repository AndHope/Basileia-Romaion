# Union of the Romanian Principalities
je_little_union = {

	icon = "gfx/interface/icons/event_icons/event_election.dds"
	
	group = je_group_foreign_affairs

	is_shown_in_lobby = {
		c:WAL ?= this
	}

	complete = {
		OR = {
			hidden_trigger = {
				AND = {
					is_player = no
					game_date >= 1859
					c:MOL = {
						is_player = no
					}
				}
			}
			AND = {
				is_at_war = no
				exists = c:MOL
				has_technology_researched = nationalism

        		NOR = {
					has_law_or_variant = law_type:law_closed_borders
				}

				c:MOL = {
					relations:ROOT >= relations_threshold:amicable
				}

				c:BYZ = {
					relations:ROOT >= relations_threshold:amicable
				}

				OR = {
					c:HRE = {
						relations:ROOT >= relations_threshold:amicable
					}
					c:DEN = {
						relations:ROOT >= relations_threshold:amicable
					}
				}

				OR = {
					c:br_tartaria = {
						relations:ROOT >= relations_threshold:amicable
					}
					c:PLC = {
						relations:ROOT >= relations_threshold:amicable
					}
				}
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = wallachia_events.3
			popup = yes
		}

		s:STATE_MOLDAVIA = {
			add_claim = ROOT
		}
	}

	weight = 9000

	should_be_pinned_by_default = yes
}
