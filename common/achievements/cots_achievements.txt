achievement_desteapta_te_romane = {
	possible = {
		country_has_primary_culture = cu:romanian
	}
	happened = {
		custom_tooltip = {
			text = achievement_desteapta_te_romane_tt
			c:ROM ?= this
		}
		custom_tooltip = {
			text = achievement_desteapta_te_romane_2_tt
			any_state = {
				filter = {
					state_region = {
						is_homeland = cu:romanian
					}
				}
				is_split_state = no
				owner = ROOT
				count = all
			}
		}
	}
}
