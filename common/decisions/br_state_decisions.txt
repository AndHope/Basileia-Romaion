remove_cossacks = {
	is_shown = {
		any_scope_state = {
			has_modifier = br_modifier_cossacks
		}
	}

	possible = {
		OR = {
            AND = {
                ig:ig_devout = {
			        is_powerful = yes
			        is_in_government = yes
		        }
		        has_insurrectionary_interest_groups = no
                country_has_state_religion = rel:catholic
				has_law_or_variant = law_type:law_national_supremacy
				NOT = { has_law_or_variant = law_type:law_freedom_of_conscience }
            }
            AND = {
                NOT = {
                    has_law_or_variant = law_type:law_hereditary_bureaucrats
                    has_law_or_variant = law_type:law_serfdom
					country_has_state_religion = rel:catholic
                }
            }
        }
		any_scope_state = {
			has_modifier = br_modifier_cossacks
		}
	}

	when_taken = {
		
		every_scope_state = {
			limit = {
				has_modifier = br_modifier_cossacks
			}
			remove_modifier = br_modifier_cossacks
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:orthodox 
				}
			}
			add_radicals = {
				religion = rel:orthodox
				value = 0.25
			}
		}

	}

	ai_chance = {
		value = 5
		if = {
			limit = {
				game_date > 1800.1.1
			}
			add = 10
		}
	}
}
