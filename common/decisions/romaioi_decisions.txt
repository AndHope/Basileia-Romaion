
fight_corruption = {
	is_shown = {
		exists = c:BYZ
		ROOT = c:BYZ
		has_modifier = romaioi_bureaucratic_corruption
	}

	possible = {
		exists = c:BYZ
		ROOT = c:BYZ
		NOT = { has_law = law_type:law_hereditary_bureaucrats }
		has_modifier = romaioi_bureaucratic_corruption
	}

	when_taken = {
		
		remove_modifier = romaioi_bureaucratic_corruption

		add_modifier = {
			name = dealing_with_corruption
			months = 52
		}
		hidden_effect = {
			trigger_event = {
				id = bulgarian_revolt.1
				days = 14
			}
	#		trigger_event = {
	#			id = vlach_rebellion.1
	#			days = 15
	#		}
		}
		add_radicals = {
			pop_type = bureaucrats
			value = large_radicals
		}

		add_radicals = {
			pop_type = aristocrats
			value = large_radicals
		}

		ig:ig_intelligentsia = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}

		ig:ig_landowners = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}

		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = romaioi_ig_angry
				months = normal_modifier_time
			}
		}
	}

	ai_chance = {
		base = 0
	}
}

end_latin_privileges = {
	is_shown = {
		exists = c:BYZ

		has_modifier = latin_privileges
	}

	possible = {
		exists = c:BYZ
		ROOT = c:BYZ
		has_law = law_type:law_interventionism
		has_law = law_type:law_protectionism
		has_modifier = latin_privileges
	}

	when_taken = {
		remove_modifier = latin_privileges
		hidden_effect = {
			trigger_event = {
				id = greek_revolt.1
				days = 14
			}
		}	
		add_modifier = {
			name = anti_latin_policies
			months = 52
		}
		change_infamy = 25

		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					culture = cu:latinoi
				}
			}
			add_radicals = {
				culture = cu:latinoi
				value = 0.75
			}
		}
	}
	ai_chance = {
		base = 0
	}
}

romaioi_themata_decision = {
	is_shown = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:elysian
		}
		has_modifier = romaioi_tagmata_primacy
	}

	possible = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:elysian
		}
		has_law = law_type:law_peasant_levies
		has_law = law_type:law_agrarianism
		has_modifier = romaioi_tagmata_primacy
		NOT = { has_modifier = romaioi_military_reorganization }
	}

	when_taken = {
		remove_modifier = romaioi_tagmata_primacy
		add_modifier = {
			name = romaioi_themata_primacy
			months = -1
		}

		add_modifier = {
			name = romaioi_military_reorganization
			months = 36
		}

		add_loyalists = {
			pop_type = aristocrats
			culture = cu:greek
			value = large_radicals
		}

		add_loyalists = {
			pop_type = farmers
			culture = cu:greek
			value = large_radicals
		}

		ig:ig_armed_forces = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}
	}

	ai_chance = {
		base = 0
	}
}
