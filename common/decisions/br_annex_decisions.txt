annex_finland = {
	is_shown = {
		exists = c:SWE
		exists = c:FIN
		ROOT = c:SWE
		c:FIN = { is_subject_of = c:SWE }
	}

	possible = {
		exists = c:SWE
		exists = c:FIN
		ROOT = c:SWE
		c:FIN = { is_subject_of = c:SWE }
		NOT = { has_law = law_type:law_hereditary_bureaucrats }
		NOT = { has_law = law_type:law_oligarchy }
		has_technology_researched = centralization
		c:FIN = {
			relations:root >= 45
		}
	}

	when_taken = {
		annex = c:FIN
		add_primary_culture = cu:finnish
		change_infamy = 10
	}

	ai_chance = {
		base = 1
	}
}
