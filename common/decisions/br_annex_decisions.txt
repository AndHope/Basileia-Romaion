annex_finland = {
	is_shown = {
		exists = c:SWE
		exists = c:FIN
		ROOT = c:SWE
		c:FIN = { is_subject_of = c:SWE }
	}

	possible = {
		exists = c:SWE
		exists = c:FIN
		ROOT = c:SWE
		c:FIN = { is_subject_of = c:SWE }

		country_rank >= rank_value:major_power

		NOT = { has_law_or_variant = law_type:law_hereditary_bureaucrats }
		NOT = { has_law_or_variant = law_type:law_oligarchy }

		has_technology_researched = centralization
		c:FIN = {
			relations:root >= 45
		}
		c:FIN = {
			is_player = no
		}
	}

	when_taken = {
		annex = c:FIN
		add_primary_culture = cu:finnish
		change_infamy = 10
	}

	ai_chance = {
		value = 5
		if = {
			limit = {
				game_date > 1750.1.1
			}
			add = 25
		}
		if = {
			limit = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			add = -1000
		}
	}
}
