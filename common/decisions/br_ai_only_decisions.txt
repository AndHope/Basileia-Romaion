		}

		trigger_event = {
			id = hungarian_revolt.1
			days = 1
		}
	}

		base = 100
	}
}

br_hre_annex_franks = {
	is_shown = {
		and = {
			exists = c:HRE
			root = c:HRE
			c:HRE = {
				is_player = no
			}
			country_has_primary_culture = cu:frankish
		}
	}

	possible = {
		is_at_war = no
		country_has_primary_culture = cu:frankish
		c:HES = {
			is_player = no
		}
		OR = {
			has_law = law_type:law_national_supremacy
			has_law = law_type:law_ethnostate
		}
	}

	when_taken = {
		remove_primary_culture = cu:frankish
		add_radicals = {
			culture = cu:frankish
			value = 0.15
		}
		add_modifier = {
			name = linguistic_discrimination
			years = -1
		}
		hidden_effect = {
			trigger_event = {
				id = frankish_revolt.1
				days = 1
			}
		}

	}

	ai_chance = {
		base = 100
	}
}
