#AI Return capital states of important countries to the AI owner if the country still exists. Decreases border gore and weird borders.

br_ai_owns_constantinople = {
	is_shown = {
		is_player = no
		exists = c:BYZ
		c:BYZ = {
			is_player = no
		}
		NOT = {
			c:BYZ ?= this
			any_scope_state = {
				state_region = s:STATE_EASTERN_THRACE
				owner = c:BYZ
			}
		}
		ROOT = {
			any_scope_state = {
				state_region = s:STATE_EASTERN_THRACE
				owner = ROOT
			}
		}	
	}

	possible = {
		is_player = no
		is_at_war = no
		c:BYZ = {
			is_player = no
		}
		hidden_trigger = {
			NOT = {
				c:BYZ ?= this
        		any_scope_state = {
					state_region = s:STATE_EASTERN_THRACE
					owner = c:BYZ
				}
			}
		}
	}

	when_taken = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_EASTERN_THRACE
			}
			set_state_owner = c:BYZ
		}
		c:BYZ = {
			set_capital = STATE_EASTERN_THRACE
			set_owes_obligation_to = {
				country = ROOT
				setting = yes
			}
		}
		ROOT = {
			change_relations = {
				country = c:BYZ
				value = 100
			}
		}
	}

	ai_chance = {
		base = 100
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}
