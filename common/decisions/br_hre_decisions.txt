
remove_noble_privileges_decision = {
	is_shown = {
		has_modifier = br_modifier_noble_privileges
	}

	possible = {
		NOT = { 
			has_law = law_type:law_hereditary_bureaucrats
			has_law = law_type:law_oligarchy
			has_law = law_type:law_traditionalism 
		}
		has_modifier = br_modifier_noble_privileges
	}

	when_taken = {
		
		remove_modifier = br_modifier_noble_privileges

		add_radicals = {
			pop_type = clergymen
			value = large_radicals
		}

		add_radicals = {
			pop_type = aristocrats
			value = very_large_radicals
		}

		ig:ig_landowners = {
			add_modifier = {
				name = br_modifier_ig_very_angry
				months = normal_modifier_time
			}
		}

	}

	#change_infamy = 50

	ai_chance = {
		value = 5
		if = {
			limit = {
				game_date > 1750.1.1
			}
			add = 10
		}
	}
}

remove_frankish_as_primary_culture_decision = {
	is_shown = {
		and = {
			exists = c:HRE
			root = c:HRE
			country_has_primary_culture = cu:frankish
		}
	}

	possible = {
		has_technology_researched = nationalism
		country_has_primary_culture = cu:frankish
		OR = {
			has_law = law_type:law_national_supremacy
			has_law = law_type:law_ethnostate
		}
	}

	when_taken = {
		remove_primary_culture = cu:frankish
		add_radicals = {
			culture = cu:frankish
			value = 0.15
		}
		add_modifier = {
			name = linguistic_discrimination
			years = -1
		}

		hidden_effect = {
			c:HES = {
				change_relations = {
					country = c:HRE
					value = -100
				}
			}
			c:WUR = {
				change_relations = {
					country = c:HRE
					value = -100
				}
			}
			c:BAD = {
				change_relations = {
					country = c:HRE
					value = -100
				}
			}
		}

	}

	ai_chance = {
		value = 0
	}
}
