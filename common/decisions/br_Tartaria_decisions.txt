revive_khazer_khanate_decision = {
	is_shown = {
		exists = c:br_tartaria
		ROOT = c:br_tartaria
		country_has_state_religion = rel:br_religion_tengri
		NOT = { country_has_state_religion = rel:jewish }
	}

	possible = {
		exists = c:br_tartaria
		ROOT = c:br_tartaria
		country_has_state_religion = rel:br_religion_tengri
		has_law_or_variant = law_type:law_monarchy
	}

	when_taken = {
		set_state_religion = rel:jewish
		remove_primary_culture = cu:mongol
		ig:ig_devout = {
			set_interest_group_name = ig_jewish_priests 
		}
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:jewish
				value = 0.25
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { is_capital = yes }
			}
			convert_population = {
				target = rel:jewish
				value = 0.15
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:jewish 
				}
			}
			add_loyalists = {
				religion = rel:jewish
				value = 0.75
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:br_religion_tengri
				}
			}
			add_radicals = {
				religion = rel:br_religion_tengri
				value = 0.25
			}
		}
		c:VEN = {
			change_relations = { 
				country = c:br_tartaria 
				value = 50 
			}
		}
		c:br_genoa = {
			change_relations = { 
				country = c:br_tartaria 
				value = 50 
			}
		}
		c:BYZ = {
			change_relations = { 
				country = c:br_tartaria 
				value = 150 
			}
		}
		c:br_great_yuan = {
			change_relations = { 
				country = c:br_tartaria 
				value = -50 
			}
		}
		c:PER = {
			change_relations = { 
				country = c:br_tartaria 
				value = -50 
			}
		}
	}

	ai_chance = {
		value = 5
	}
}

#Decision to make it easier for a Yuan Player win the starting war.
br_mongol_army_prowess_decision = {
	is_shown = {
		OR = {
			country_has_primary_culture = cu:tatar
			country_has_primary_culture = cu:mongol
		}
		NOT = { has_modifier = br_modifier_horde_army_prowess }
	}

	possible = {
		OR = {
			country_has_primary_culture = cu:tatar
			country_has_primary_culture = cu:mongol
		}
		NOT = { has_modifier = br_modifier_horde_army_prowess }
		is_player = yes
	}

	when_taken = {
		add_modifier = {
			name = br_modifier_horde_army_prowess
			years = 5
		}
		activate_law = law_type:law_mixed_army
	}

	ai_chance = {
		value = 0
	}
}
