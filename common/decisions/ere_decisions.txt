fight_corruption = {
	is_shown = {
		exists = c:BYZ
		ROOT = c:BYZ
		has_modifier = romaioi_bureaucratic_corruption
	}

	possible = {
		exists = c:BYZ
		ROOT = c:BYZ
		NOT = { has_law = law_type:law_hereditary_bureaucrats }
		has_modifier = romaioi_bureaucratic_corruption
	}

	when_taken = {
		
		remove_modifier = romaioi_bureaucratic_corruption

		add_modifier = {
			name = dealing_with_corruption
			months = 52
		}
		hidden_effect = {
			trigger_event = {
				id = bulgarian_revolt.1
				days = 14
			}
	#		trigger_event = {
	#			id = vlach_rebellion.1
	#			days = 15
	#		}
		}
		add_radicals = {
			pop_type = bureaucrats
			value = large_radicals
		}

		add_radicals = {
			pop_type = aristocrats
			value = large_radicals
		}

		ig:ig_intelligentsia = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}

		ig:ig_landowners = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}

		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = romaioi_ig_angry
				months = normal_modifier_time
			}
		}
	}

	ai_chance = {
		base = 0
	}
}

	}

	when_taken = {
		remove_modifier = latin_privileges
		activate_law = law_type:law_state_religion

			}

			name = anti_latin_policies
			months = 52
		}
		change_infamy = 25

		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = romaioi_ig_very_angry
				months = normal_modifier_time
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					culture = cu:latinoi
				}
			}
			add_radicals = {
				culture = cu:latinoi
				value = 0.75
			}
		}
	}
	ai_chance = {
		base = 0
	}
}

chariot_racing_decision = {
	is_shown = {
		exists = c:BYZ
		ROOT = c:BYZ
		NOT = {
			has_modifier = modifier_chariot_racing
		}
	}
	
	possible = {
		any_scope_building = {
			is_building_type = building_hippodrome
		}
	}
	
	when_taken = {
		add_modifier = {
			name = modifier_chariot_racing
		}
		custom_tooltip = {
			text = chariot_racing_enabled_tt	
			trigger_event = {
				id = chariot_racing.1
			}
		}
	}
}
