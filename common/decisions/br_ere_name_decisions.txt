br_show_ere_name_decisions = {
	is_shown = {
		exists = c:BYZ
		c:BYZ = ROOT
		is_player = yes
		
		NOT = {
			has_variable = showing_latinized_names
		}
	}

	possible = {
		is_at_war = no
	}

	when_taken = {
		set_variable = showing_latinized_names
	}

	ai_chance = {
		value = 0
	}
}

br_hide_ere_name_decisions = {
	is_shown = {
		exists = c:BYZ
		c:BYZ = ROOT
		is_player = yes
		has_variable = showing_latinized_names
	}

	possible = {
		is_at_war = no
	}

	when_taken = {
		remove_variable = showing_latinized_names
	}

	ai_chance = {
		value = 0
	}
}

ere_re_name_decision = {
	is_shown = {
		exists = c:BYZ
		c:BYZ = ROOT
		is_player = yes
		has_variable = showing_latinized_names
		NOR = {
			has_variable = re_name_chosen_var
		}
	}

	possible = {
		is_at_war = no
	}

	when_taken = {
		set_variable = re_name_chosen_var
		custom_tooltip = {
			text = ere_re_name_tt
		}
	}

	ai_chance = {
		value = 0
	}
}
