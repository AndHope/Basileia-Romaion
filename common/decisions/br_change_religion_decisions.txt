revive_olympic_games_decision = {
	is_shown = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:elysian
			country_has_state_religion = rel:br_religion_hellenic
		}
	}

	possible = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:elysian
			country_has_state_religion = rel:br_religion_hellenic
		}
		has_technology_researched = organized_sports
		hidden_trigger = {
			NOT = { has_variable = revive_olympics_decision_taken }
		}
		ig:ig_devout = {
			is_in_government = no
			#is_marginal = no
		}
		NOT = { has_law = law_type:law_religious_schools }
	}
	
	when_taken = {
		set_variable = {
			name = revive_olympics_decision_taken
			value = yes
		}
		add_modifier = {
			name = modifier_olympic_games
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
			}
			convert_population = {
				target = rel:br_religion_hellenic
				value = 0.05
			}
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = { bureaucracy > 0 }
			add = 5
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}		
	}
}

hellenic_decision = {
	is_shown = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:elysian
		}
		NOT = { country_has_state_religion = rel:br_religion_hellenic }
		has_variable = revive_olympics_decision_taken
	}

	possible = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:elysian
		}
		ig:ig_devout = {
			is_in_government = no
			#is_marginal = no
		}
		#NOT = { has_modifier = br_modifier_religious_chaos }
		has_law = law_type:law_freedom_of_conscience
		NOT = { has_law = law_type:law_religious_schools }
		has_law = law_type:law_monarchy
	}

	when_taken = {
		set_state_religion = rel:br_religion_hellenic
		ig:ig_devout = {
			set_interest_group_name = ig_neo_hellenes
			remove_ideology = ideology_patriarchal
			add_ideology = ideology_hellenic_clergy
		}
		#add_modifier = {
		#	name = br_modifier_religious_chaos
		#	months = 52
		#}		
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:br_religion_hellenic
				value = 0.25
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { is_capital = yes }
			}
			convert_population = {
				target = rel:br_religion_hellenic
				value = 0.15
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:br_religion_hellenic
				}
			}
			add_loyalists = {
				religion = rel:br_religion_hellenic
				value = 0.50
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:orthodox 
				}
			}
			add_radicals = {
				religion = rel:orthodox
				value = 0.50
			}
		}
		change_relations = { 
			country = c:HRE 
			value = -150 
		}
		change_relations = { 
			country = c:PAP 
			value = -150 
		}
	}
