
abolish_feudalism_decision = {
	is_shown = {
		has_social_hierarchy = br_feudalism_social_hierarchy
		has_modifier = br_modifier_feudalism
	}

	possible = {
		NOT = { 
			has_law = law_type:law_hereditary_bureaucrats
			has_law = law_type:law_oligarchy
			has_law = law_type:law_traditionalism
			has_law = law_type:law_serfdom
			has_law = law_type:br_law_feudalism_enforced
			has_law = law_type:br_law_feudalism_codified
		}
		has_social_hierarchy = br_feudalism_social_hierarchy
		has_modifier = br_modifier_feudalism
	}

	when_taken = {
		
		set_social_hierarchy = default_social_hierarchy
		remove_modifier = br_modifier_feudalism

		add_radicals = {
			pop_type = clergymen
			value = medium_radicals
		}

		add_radicals = {
			pop_type = aristocrats
			value = large_radicals
		}

		ig:ig_landowners = {
			add_modifier = {
				name = br_modifier_ig_very_angry
				days = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = br_modifier_ig_very_happy
				days = normal_modifier_time
			}
		}


		#Remove feudalism ideologies to IG's
		every_interest_group = {
			remove_ideology = ideology_pro_feudalism
			remove_ideology = ideology_anti_feudalism
		}
	}

	ai_chance = {
		value = 1
		if = {
			limit = {
				game_date > 1800.1.1
			}
			add = 10
		}
	}
}
