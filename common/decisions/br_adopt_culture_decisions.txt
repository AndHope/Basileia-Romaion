armenian_decision = {
	is_shown = {
		exists = c:BYZ
		ROOT = c:BYZ
		NOT = { country_has_primary_culture = cu:armenian }
	}

	possible = {
		exists = c:BYZ
		ROOT = c:BYZ
		NOT = {
			has_law = law_type:law_national_supremacy 
			has_law = law_type:law_ethnostate 
			country_has_primary_culture = cu:armenian
		}
	}

	when_taken = {
		add_primary_culture = cu:armenian
		remove_modifier = br_modifier_romaioi_culture_supremacy

		hidden_effect = {
			trigger_event = {
				id = anatolian_revolt.2
				days = 14
			}
		}

		add_loyalists = {
			culture = cu:armenian
			value = 0.25
		}
		ig:ig_landowners = {
			add_modifier = {
				name = br_modifier_ig_angry
				months = normal_modifier_time
			}
		}
	}

	ai_chance = {
		base = 0
	}
}
