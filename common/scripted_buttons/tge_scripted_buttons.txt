tge_select_deity_button = {
	name = "tge_select_deity_button"
	desc = "tge_select_deity_button_desc"

	visible = {
		always = yes
	}
	possible = {
		custom_tooltip = {
			text = tge_has_deity_cooldown_NOT
			NOT = {
				has_variable = tge_deity_cooldown_var
			}
		}
		NOR = {
			country_has_state_religion = rel:br_tge_religion_teotl
			hidden_trigger = {
				has_variable = tge_selecting_deity_var
			}
		}
	}
	effect = {
		set_variable = {
			name = tge_selecting_deity_var
			months = 3
		}
		trigger_event = {
			id = tge_tonalism.010
			popup = yes
		}
	}
}

tge_revive_bloodletting_button = {
	name = "tge_revive_bloodletting_button"
	desc = "tge_revive_bloodletting_button_desc"

	visible = {
		NOT = {
			has_modifier = modifier_tge_ritual_bloodletting
		}
	}
	possible = {
		has_law = law_type:law_state_religion
		NOR = {
			has_law = law_type:law_slavery_banned
			hidden_trigger = {
				has_variable = tge_revival_bloodletting_var
			}
		}
		tge_is_venerating_huitzilopochtli_trigger = yes
		infamy >= infamy_threshold:pariah
	}
	effect = {
		set_variable = {
			name = tge_revival_bloodletting_var
			months = 12
		}
		trigger_event = {
			id = tge_tonalism.040 # Bloodletting Revived
		}
		show_as_tooltip = {
			add_modifier = modifier_tge_ritual_bloodletting
		}
	}
}
