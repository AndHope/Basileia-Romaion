# UI can obtain a datamodel with all the scripted buttons via "[JournalEntry.GetScriptedButtons]". Each scripted button can be referred to via "ScriptedButton".
# Scripted buttons' properties can be accessed by UI via
# - [ScriptedButton.GetName] -> provides a string
# - [ScriptedButton.GetDesc] -> provides a string
# - [ScriptedButton.GetEffectDesc] -> provides a string
# - [ScriptedButton.IsVisible] -> provides a bool
# - [ScriptedButton.IsPossible] -> provides a bool
# - [ScriptedButton.ExecuteEffect] -> provides nothing, executes the effect

je_create_princely_league_button = {
	name = "je_create_princely_league_button"
	desc = "je_create_princely_league_button_desc"

	#TODO write custom tooltips for the buttons effect

	visible = { always = yes }

	#Country scope
	ai_chance = {
		value = 0
		if = {
			limit = {
				country_rank >= rank_value:major_power
			}
			add = 5
		}
	}

	possible = {
		# custom_tooltip = {
		# 	text = princely_league_not_formed_tt
		# 	NOT = { has_variable = princely_league_formed }
		# }
	}

	effect = {

		set_variable = created_princely_league #variable for the country that formed the league
		
		add_modifier = {
			name = br_modifier_princely_league_leader
			years = -1
		}

		root = {
			save_scope_as = league_leader
		}

		overlord = {
			create_unidirectional_truce = {
				country = root
				months = 60
			} 
		}

		custom_tooltip = {
			text = princely_league_being_formed_tt
			every_country = {
				limit = {
					country_is_in_europe = yes #every european state gets notefied
					# NOT = {
					# 	this = root
					# }
				}
				set_variable = princely_league_formed
				if = {
					limit = { 
						overlord = c:HRE
						not = { this = c:HRE }
				 	}
					trigger_event = { id = princely_league_formation.2 popup = yes } #event for League candidates
				}
				else = {
					trigger_event = { id = princely_league_formation.1 popup = yes } #event for bystanders
				}
			}
		}

		add_journal_entry = {
			type = je_princely_league_main
		}

		#Create the Situation JE for the Princely League
		# create_com_situation_journal_entry = {
		# 	journal = je_princely_league_main
		# 	situation_id = br_princely_league_situation_id
		# 	icon = com_icon_rise_of_communism	#temp
		# 	side_leader_country_left = c:FLA
		# 	side_name_left = br_princely_league_left_side_name
		# 	side_desc_left = br_princely_league_left_side_desc
		# 	side_leader_country_right = c:HRE
		# 	side_name_right = br_princely_league_right_side_name
		# 	side_desc_right = br_princely_league_right_side_desc
		# }
	}
}

br_je_invite_members_to_league_button = {
	name = "je_invite_members_to_league_button"
	desc = "je_invite_members_to_league_button_desc"

	#TODO write custom tooltips for the buttons effect

	visible = { always = yes }

	cooldown = {
		months = 1
	}

	possible = {
		# not = {
		# 	has_variable = inviting_members_to_league
		# }
		has_modifier = br_modifier_princely_league_leader
		custom_tooltip = {
			text = formed_princely_league_tt
			has_variable = created_princely_league
		}
	}

	effect = {
		# set_variable = inviting_members_to_league
		
		root = {
			save_scope_as = league_leader
		}

		# overlord = {
		# 	create_unidirectional_truce = {  
		# 		country = root
		# 		months = 36
		# 	} 
		# }

		every_country = {
			limit = {
				country_is_in_europe = yes
				is_subject = yes
				NOT = { this = root }
				overlord = c:HRE
				liberty_desire > 75
				NOT = {
					br_is_county_supporting_independence_scripted_trigger = {
						INITIATOR_COUNTRY = scope:league_leader
						TARGET_COUNTRY = prev
					}
				}
			}
			trigger_event = { id = princely_league_invitation.1 popup = yes }
		}
	}

	#Country scope
	ai_chance = {
		value = 0
		if = {
			limit = {
				c:FLA = { liberty_desire >= 50 }
			}
			add = 5
		}
	}
}

br_je_princely_league_main_join_left_side_button = {
	name = "br_je_princely_league_main_join_left_side_button"
	desc = "br_je_princely_league_main_join_left_side_button_desc"

	#TODO write custom tooltips for the buttons effect

	visible = { always = yes }

	cooldown = {
		months = 1
	}

	possible = {
		not = {
			or = {
				is_com_situation_side_leader = {
					side = left
				}
				is_com_situation_side_leader = {
					side = right
				}
				is_com_situation_faction_member = {
					side = left
					faction = br_princely_league_left_faction_members
				}
			}
		}
	}

	effect = {
		#country joins the princely league faction
		save_temporary_scope_as = br_princely_league_new_left_side_member_scope

		#Remove them from the previous faction (All countries may only belong to one faction)
		remove_com_situation_faction_country_all = {
			country = scope:br_princely_league_new_left_side_member_scope
		}

		add_com_situation_faction_country = {
			side = left
			name = br_princely_league_left_faction_members
			country = scope:br_princely_league_new_left_side_member_scope
		}

		#By default when you join your stance is set to autonomy
		br_princely_league_main_left_side_member_change_stance_to_autonomy = yes
	}

	#Country scope
	ai_chance = {
		value = 0
		if = {
			limit = {
				"relations(scope:overlord)" < relations_threshold:neural
				"relations(scope:overlord)" >= relations_threshold:poor
			}
			add = 5
		}
		else_if = {
			limit = {
				"relations(scope:overlord)" < relations_threshold:poor
				"relations(scope:overlord)" >= relations_threshold:cold
			}
			add = 10
		}
		else_if = {
			limit = {
				"relations(scope:overlord)" < relations_threshold:cold
				"relations(scope:overlord)" >= relations_threshold:hostile
			}
			add = 25
		}
	}
}
