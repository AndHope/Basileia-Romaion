country_can_form_power_bloc = {
    is_revolutionary = no
    
    trigger_if = {
        limit = {
            NOT = {
                is_country_type = unrecognized
            }
        }
        #country_rank >= rank_value:major_power
        is_country_type = recognized
            prestige >= { 
                add = {
                    ordered_country = {
                        order_by = prestige
                        position = 0
                        
                        add = prestige
                        multiply = 0.1
                    }
                }
            }
    }
    trigger_else = {
        country_rank >= rank_value:unrecognized_major_power
        OR = {
            is_country_type = recognized # Just to make it clear to the user that it's an option
            prestige >= { # Within 25% of prestige of top unrecognized country
                add = {
                    ordered_country = {
                        limit = {
                            is_country_type = unrecognized
                        }
                        order_by = prestige
                        position = 0
                        
                        add = prestige
                        multiply = 0.50
                    }
                }
            }
        }
    }
}
