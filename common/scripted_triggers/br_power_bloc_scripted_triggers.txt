country_can_form_power_bloc = {
    is_revolutionary = no
    
    trigger_if = {
        limit = {
            is_country_type = recognized
        }
        #country_rank >= rank_value:major_power
        prestige >= { # Within 90% of prestige of top recognized country
            add = {
                ordered_country = {
                    order_by = prestige
                    position = 0
                    
                    add = prestige
                    multiply = 0.1
                }
            }
        }
    }
    trigger_else_if = {
        limit = {
            or = {
                is_country_type = colonial
                is_country_type = company
            }
        }
        prestige >= { # Within 85% of prestige of top recognized country
            add = {
                ordered_country = {
                    order_by = prestige
                    position = 0
                    
                    add = prestige
                    multiply = 0.15
                }
            }
        }
    }
    trigger_else = {
        country_rank >= rank_value:unrecognized_regional_power
        OR = {
            is_country_type = recognized # Just to make it clear to the user that it's an option
            prestige >= { # Within 75% of prestige of top recognized country
                add = {
                    ordered_country = {
                        limit = {
                            is_country_type = unrecognized
                        }
                        order_by = prestige
                        position = 0
                        
                        add = prestige
                        multiply = 0.25
                    }
                }
            }
        }
    }
}
