br_is_county_supporting_independence_scripted_trigger = {
	$INITIATOR_COUNTRY$ = {
		any_scope_treaty = {
			binds = $TARGET_COUNTRY$
			any_scope_article = {
				has_type = support_independence
			}
		}
	}
}
