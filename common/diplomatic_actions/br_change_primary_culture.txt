change_primary_culture = {
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		nationalism
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
			is_direct_subject_of = root
		}
		any_primary_culture = {
			NOT = { is_primary_culture_of = scope:target_country }
		}
		
		is_player = yes
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
		NOT = { has_war_with = scope:target_country }
		scope:target_country.relations:root > relations_threshold:cordial
		has_diplomatic_relevance = scope:target_country
	}
	
	pact = {	
		cost = 50
	
		is_two_sided_pact = no
		show_in_outliner = yes

		income_transfer = 0.10
		recipient_gets_income_transfer = yes
		income_transfer_based_on_recipient = yes

		monthly_effect = {
			custom_tooltip = {
				text = change_primary_culture_monthly_effect
				random_list = { 
					19 = {}
					1 = { 
						root = {
							trigger_event = {
					            id = ib_events.01
					        }
					    }
					}
				}								
			}
		} 	
	
		is_about_to_auto_break = {
            scope:target_country.relations:root <= 23 # this: + 3
        }

		should_auto_break = {
			scope:target_country.relations:root <= relations_threshold:cordial
		}		
	
		should_invalidate = {
			OR = {
				has_war_with = scope:target_country				
				is_country_type = decentralized
				scope:target_country = {
					OR = {
						is_country_type = decentralized
						NOT = { is_direct_subject_of = root }
					}
				}
				NOT = {
					any_primary_culture = {
						NOT = { is_primary_culture_of = scope:target_country }
					}
				}
				in_default = yes
			}
		}
	}
}
