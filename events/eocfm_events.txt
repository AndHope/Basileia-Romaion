eocfm.012 = { # A Heretical Ruler
	type = country_event
	placement = root
	
	title = eocfm.012.t
	desc = eocfm.012.d
	flavor = eocfm.012.f
	
	event_image = {
		video = "votp_monarch_holding_court"
	}
	
	icon = "gfx/interface/icons/event_icons/event_orthodox.dds"
	
	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	duration = 3
	
	dlc = dlc_eocfm
	
	trigger = {
		NOT = {
			has_variable = eocfm_ruler_conversion_considered
		}
		has_law = law_type:law_monarchy
		ruler = {
			NOT = {
				has_religion = orthodox
			}
		}
		heir ?= {
			NOT = {
				has_religion = orthodox
			}
		}
		ig:ig_devout = {
			ig_approval <= 5
		}
	}
	
	immediate = {
		set_variable = eocfm_ruler_conversion_considered
		ruler = {
			save_scope_as = eocfm_ruler_scope
		}
		heir = {
			save_scope_as = eocfm_heir_scope
		}
	}	
	
	option = { # Both will convert
		name = eocfm.012.a
		default_option = yes
		scope:eocfm_ruler_scope = {
			change_character_religion = rel:orthodox
			add_modifier = modifier_eocfm_accepted_conversion
		}
		scope:eocfm_heir_scope = {
			change_character_religion = rel:orthodox
			add_modifier = modifier_eocfm_accepted_conversion
		}
		add_modifier = {
			name = modifier_eocfm_ruler_accepted_conversion
			months = normal_modifier_time
		}
		add_loyalists = {
			religion = rel:orthodox
			value = medium_radicals
		}
	}
	
	option = { # Heir will convert
		name = eocfm.012.b
		scope:eocfm_ruler_scope = {
			add_modifier = {
				name = modifier_eocfm_refused_conversion
				months = very_long_modifier_time
			}
		}
		scope:eocfm_heir_scope = {
			change_character_religion = rel:orthodox
			add_modifier = modifier_eocfm_accepted_conversion
		}
		add_loyalists = {
			religion = rel:orthodox
			value = small_radicals
		}
	}
	
	option = { # 99 THESES GO BRRRRRRRRR

			add_modifier = {
				name = modifier_eocfm_refused_conversion

			}
		}

eocfm.040 = { # Priestly Excesses
	type = country_event
	placement = root
	
	title = eocfm.040.t

	flavor = eocfm.040.f
	
	event_image = {
		video = "southamerica_christianity"
	}

	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	duration = 3
	
	dlc = dlc_eocfm
	
	trigger = {
		NOT = {
			has_variable = eocfm_decree_event_var
		}
		any_scope_state = {
			has_decree = decree_eocfm_consecrate_metropolitan
			NOT = {
				has_modifier = modifier_eocfm_grand_cathedral
			}
		}
	}
	
	immediate = {
		random_scope_state = {
			limit = {
				has_decree = decree_eocfm_consecrate_metropolitan
				NOT = {
					has_modifier = modifier_eocfm_grand_cathedral
				}
			}
			save_scope_as = eocfm_decree_state_scope
		}
		set_variable = {
			name = eocfm_decree_event_var
			days = 1825 # Five year lockout
		}
	}
	
	option = { # Allow Construction
		name = eocfm.040.a
		default_option = yes
		scope:eocfm_decree_state_scope = {
			add_modifier = {
				name = modifier_eocfm_new_cathedral
				months = long_modifier_time
			}
			add_loyalists_in_state = {
				pop_type = clergymen
				value = medium_radicals
			}
			add_radicals_in_state = {
				pop_type = academics
				value = small_radicals
			}
			add_radicals_in_state = {
				pop_type = clerks
				value = small_radicals
			}
		}
	}
	
	option = { # Curtail Construction
		name = eocfm.040.b
		scope:eocfm_decree_state_scope = {
			add_radicals_in_state = {
				pop_type = clergymen
				value = medium_radicals
			}
			add_loyalists_in_state = {
				pop_type = academics
				value = medium_radicals
			}
			add_loyalists_in_state = {
				pop_type = clerks
				value = medium_radicals
			}
		}
	}
	
	option = { # Give More Moni
		name = eocfm.040.c
		add_treasury = -50000
		scope:eocfm_decree_state_scope = {
			add_modifier = {
				name = modifier_eocfm_grand_cathedral
				months = long_modifier_time
			}
		}

eocfm.140 = { # Seat Integration Disputed (Part 1)
	type = country_event
	placement = root
	
	title = eocfm.140.t
	desc = eocfm.140.d
	flavor = eocfm.140.f
	
	event_image = {
		video = "unspecific_politicians_arguing"
	}
	
	icon = "gfx/interface/icons/event_icons/event_orthodox.dds"
	
	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	duration = 3
	
	dlc = dlc_eocfm
	
	trigger = {
		eocfm_controls_any_seat_trigger = yes
		any_country = {
			NOT = {
				this = root
			}
			has_variable = eocfm_integration_var
			var:eocfm_integration_var < 80
			NOR = {
				has_modifier = modifier_eocfm_asserted_integration
				has_modifier = modifier_eocfm_folded_integration
			}
			relations:root < relations_threshold:cordial # Usually means that only Russia will get in your way most playthroughs, but this is historically pertinent
		}
		NOT = {
			has_global_variable = eocfm_seat_dispute_cooldown_var
		}
	}
	
	immediate = {
		set_global_variable = {
			name = eocfm_seat_dispute_cooldown_var
			months = 12 # lazy way to block event from running itself over
		}
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				has_variable = eocfm_integration_var
				var:eocfm_integration_var < 100
				var:eocfm_integration_var > 24
				NOR = {
					has_modifier = modifier_eocfm_asserted_integration
					has_modifier = modifier_eocfm_folded_integration
				}
			}
			save_scope_as = eocfm_integrator_scope
		}
		set_variable = eocfm_seat_dispute_starter
	}
	
	option = { # Challenge their seat integration
		name = eocfm.140.a
		default_option = yes
		change_relations = {
			country = scope:eocfm_integrator_scope
			value = -20
		}
		scope:eocfm_integrator_scope = {
			set_variable = eocfm_seat_opposed
		}
	}
	
	option = { # Support their seat integration
		name = eocfm.140.b
		change_relations = {
			country = scope:eocfm_integrator_scope
			value = 20
		}
	}
	
	after = {
		scope:eocfm_integrator_scope = {
			trigger_event = {
				id = eocfm.141 # Seat Integration Disputed (Part 2)
			}
		}
	}

}
