namespace = veiled_exarchate

# For Great Power
veiled_exarchate.1 = {
	type = country_event
	placement = root
	title = veiled_exarchate.1.t
	desc = veiled_exarchate.1.d
	flavor = veiled_exarchate.1.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/middleeast_middleclass_cafe.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/middleclass_cafe"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
	}

	immediate = {
		save_scope_as = aegypt_overlord_country
		c:br_aegypt = {
			save_scope_as = aegypt_country
			ruler = {
				save_scope_as = aegypt_ruler
			}
		}
		c:BYZ = {
			save_scope_as = byzantine_country
			ruler = {
				save_scope_as = byzantine_ruler
			}
		}
	}

	option = {
		name = veiled_exarchate.1.a
		default_option = yes
		add_modifier = {
			name = veiled_influence
			months = normal_modifier_time
		}
	}

	option = {
		name = veiled_exarchate.1.b
		c:br_aegypt = {
			set_state_religion = rel:orthodox
			add_modifier = {
				name = br_modifier_exarchate
				months = -1
			}
			add_modifier = {
				name = subservient_patriarch
				years = 50
			}
		#	remove_modifier = {
		#		name = autocephalous_patriarchate_alexandria
		#	}
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
		scope:aegypt_ruler = {
			kill_character = yes
		}
		change_relations = {
			country = c:br_aegypt
			value = 50
		}
		c:br_kyrenaica = {
			set_state_religion = rel:orthodox
			add_modifier = {
				name = br_modifier_exarchate
				months = -1
			}
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
		c:br_romaioi_india_company = {
			add_primary_culture = cu:greek
			set_state_religion = rel:orthodox
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
		c:br_romaioi_india_company = {
			set_state_religion = rel:orthodox
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
	}
}
