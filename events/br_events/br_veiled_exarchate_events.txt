namespace = veiled_exarchate

veiled_exarchate.1 = {
	type = country_event
	placement = root
	title = veiled_exarchate.1.t
	desc = veiled_exarchate.1.d
	flavor = veiled_exarchate.1.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/middleeast_middleclass_cafe.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/middleclass_cafe"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
	}

	immediate = {
		save_scope_as = aegypt_overlord_country
		c:br_aegypt = {
			save_scope_as = aegypt_country
			ruler = {
				save_scope_as = aegypt_ruler
			}
		}
		c:BYZ = {
			save_scope_as = byzantine_country
			ruler = {
				save_scope_as = byzantine_ruler
			}
		}
	}

	option = {
		name = veiled_exarchate.1.a
		default_option = yes
		change_infamy = -5
		c:br_aegypt = {
			add_modifier = {
				name = br_modifier_autonomous_march
				years = 50
			}
		}
		change_relations = {
			country = c:br_aegypt
			value = 50
		}
		change_relations = {
			country = c:br_kyrenaica
			value = 25
		}
		change_relations = {
			country = c:br_romaioi_india_company
			value = 25
		}
		create_diplomatic_pact = {
			country = c:br_aegypt	
			type = grant_own_market
		}
	}

	option = {
		name = veiled_exarchate.1.b
		change_infamy = 5
		c:br_aegypt = {
			set_state_religion = rel:orthodox
			ig:ig_devout = {
				hidden_effect = { set_interest_group_name = ig_orthodox_patriarch }
			}
			add_modifier = {
				name = br_modifier_exarchate
				months = -1
			}
			capital = {
				convert_population = {
					target = rel:orthodox
					value = 0.1
				}
			}
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
		scope:aegypt_ruler = {
			kill_character = yes
		}
		c:br_kyrenaica = {
			set_state_religion = rel:orthodox
			ig:ig_devout = {
				hidden_effect = { set_interest_group_name = ig_orthodox_patriarch }
			}
			add_modifier = {
				name = br_modifier_exarchate
				months = -1
			}
			capital = {
				convert_population = {
					target = rel:orthodox
					value = 0.1
				}
			}
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
		c:br_romaioi_india_company = {
			add_primary_culture = cu:greek
			set_state_religion = rel:orthodox
			ig:ig_devout = {
				hidden_effect = { set_interest_group_name = ig_orthodox_patriarch }
			}
			capital = {
				convert_population = {
					target = rel:orthodox
					value = 0.1
				}
			}
			every_scope_state = {
				limit = {
					owner = ROOT
					is_incorporated = yes
				}
				convert_population = {
					target = rel:orthodox
					value = 0.05
				}
			}
			if = {
				limit = {
					any_scope_pop = { 
						religion = rel:oriental_orthodox 
					}
				}
				add_radicals = {
					religion = rel:oriental_orthodox
					value = 0.1
				}
			}
		}
		change_relations = {
			country = c:VEN
			value = -25
		}
	
	}
}
