#event for NGF
treaty_of_poznan_events.1 = {
	type = country_event
	placement = ROOT
	
	title = treaty_of_poznan_events.1.t
	desc = treaty_of_poznan_events.1.d
	flavor = treaty_of_poznan_events.1.f

	cooldown = { months = normal_modifier_time }
	
	event_image = {
		video = "southamerica_aristocrats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	duration = 3

	trigger = {
		c:NGF ?= this
		NOT = {
			has_variable = treaty_of_poznan_initiation_var
		}
		year >= 1738
	}

	immediate = {
		set_variable = treaty_of_poznan_initiation_var
		c:PLC = {
			save_scope_as = br_commonwealth_scope
		}
	}

	option = {
		name = treaty_of_poznan_events.1.a
		default_option = yes
		ai_chance = {
			base = 95
		}
		c:PLC ?= {
			trigger_event = { id = treaty_of_poznan_events.2 popup = yes } 
		}
		custom_tooltip = if_they_accept_tt
		custom_tooltip = treaty_of_poznan_explanation_tt
		show_as_tooltip = {
			add_journal_entry = {
				type = br_je_conference_of_poznan
			}
			c:PLC = {
				add_journal_entry = {
					type = br_je_conference_of_poznan
				}
			}
			every_country = {
				limit = {
					country_rank >= rank_value:great_power
					NOR = { 
						root ?= this
						c:PLC ?= this
					}
					country_is_in_europe = yes
				}
				trigger_event = { id = treaty_of_poznan_events.3 popup = yes } 
			}
		}
	}


	option = {
		name = treaty_of_poznan_events.1.b
		ai_chance = {
			base = 5
		}
		c:PLC = {
			change_relations = { country = root value = -80 }
			create_diplomatic_catalyst = {  
				type = catalyst_event_negative
				target = root
			}
		}
		every_country = {
			limit = {
				country_rank >= rank_value:great_power
				NOT = { this = root }
				country_is_in_europe = yes
			}
			change_relations = { country = root value = -40 }
		}
		hidden_effect = {
			set_secret_goal = {
				country = c:PLC
				secret_goal = conquer
			}
			c:PLC = {
				set_secret_goal = {
					country = c:PRU
					secret_goal = conquer
				}
			}
		}
	}
}

				exists = c:GER

				save_scope_as = br_north_german_federation_scope
			}
		}
		else_if = {
			limit = {
				exists = c:NGF
			}

			save_scope_as = br_commonwealth_scope
		}
	}

		name = treaty_of_poznan_events.7.a

			limit = { 
				not = { has_variable != br_propose_independence_of_prussia_var }
			}

				c:NGF ?= {
					create_diplomatic_pact = {
						country = c:NGF
						type = puppet
					}
				}

				br_remove_prussian_fate_variables_scripted_effect = yes #remove event options var that are mutualy exclusive (ex: pru can not gain indendence and be subjugated)

		}

			hidden_effect = {
				remove_variable = br_propose_independence_of_prussia_var

			trigger_event = { id = treaty_of_poznan_events.7 popup = yes } #this is to trigger the same event so you can also chose the other option, but it does not seem work
		}
	}

			}
			show_as_tooltip = {
				c:NGF ?= {
					create_diplomatic_pact = {
						country = c:NGF
						type = puppet
					}
				}
			}
			hidden_effect = {
				br_remove_prussian_fate_variables_scripted_effect = yes
				set_variable = br_propose_plc_subjugation_of_prussia_var
			}
		}
		else = {
			hidden_effect = {
				remove_variable = br_propose_plc_subjugation_of_prussia_var
			}
		}
		hidden_effect = {

		}

	}
