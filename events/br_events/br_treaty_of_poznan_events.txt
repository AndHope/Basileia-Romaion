#The fate of Prussia
treaty_of_poznan_events.8 = {
	type = country_event
	placement = ROOT
	
	title = treaty_of_poznan_events.8.t
	desc = treaty_of_poznan_events.8.d
	flavor = treaty_of_poznan_events.8.f
	
	event_image = {
		video = "africa_leader_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	duration = 3

	trigger = {
	}

	immediate = {
		set_variable = br_deciding_the_fate_of_prussia_button_var
		if = {
			limit = {
				exists = c:GER
			}
			c:GER ?= {
				save_scope_as = br_north_german_federation_scope
			}
		}
		else_if = {
			limit = {
				exists = c:NGF
			}
			c:NGF ?= {
				save_scope_as = br_north_german_federation_scope
			}
		}
		c:PLC ?= {
			save_scope_as = br_commonwealth_scope
		}
	}

	#prussia becomes independent
	option = {
		name = treaty_of_poznan_events.8.a
		default_option = yes
		if = {
			limit = { 
				not = { has_variable = br_propose_independence_of_prussia_var }
			}
			show_as_tooltip = {
				br_prussia_independent_scripted_effect = yes
			}
			hidden_effect = {
				br_remove_prussian_fate_variables_scripted_effect = yes #remove event options var that are mutualy exclusive (ex: pru can not gain indendence and be subjugated)
				set_variable = br_propose_independence_of_prussia_var
			}
		}
		else = {
			hidden_effect = {
				remove_variable = br_propose_independence_of_prussia_var
			}
		}
		hidden_effect = {
			trigger_event = { id = treaty_of_poznan_events.8 popup = yes } #this is to trigger the same event so you can also chose the other option, but it does not seem work
		}
	}

	#plc puppets prussia
	option = {
		name = treaty_of_poznan_events.8.b
		default_option = yes
		if = {
			limit = { 
				not = { has_variable = br_propose_plc_subjugation_of_prussia_var }
			}
			show_as_tooltip = {
				br_plc_subjugation_of_prussia_var_scripted_effect = yes
			}
			hidden_effect = {
				br_remove_prussian_fate_variables_scripted_effect = yes
				set_variable = br_propose_plc_subjugation_of_prussia_var
			}
		}
		else = {
			hidden_effect = {
				remove_variable = br_propose_plc_subjugation_of_prussia_var
			}
		}
		hidden_effect = {
			trigger_event = { id = treaty_of_poznan_events.8 popup = yes } #this is to trigger the same event so you can also chose the other option, but it does not seem work
		}
	}

	#ngf puppets prussia
	option = {
		name = treaty_of_poznan_events.8.c
		default_option = yes
		if = {
			limit = { 
				not = { has_variable = br_propose_ngf_subjugation_of_prussia_var }
			}

		}
	}
	# .d and .f can not be used as they are used for discription an flavor

	#plc annexes prussia
	option = {
		name = treaty_of_poznan_events.8.e
		default_option = yes
		if = {
			limit = { 
				not = { has_variable = br_propose_plc_annexation_of_prussia_var }
			}
			show_as_tooltip = {
				br_plc_annexation_of_prussia_scripted_effect = yes
			}
			hidden_effect = {
				br_remove_prussian_fate_variables_scripted_effect = yes
				set_variable = br_propose_plc_annexation_of_prussia_var
			}
		}
		else = {
			hidden_effect = {
				remove_variable = br_propose_plc_annexation_of_prussia_var
			}
		}
		hidden_effect = {
			trigger_event = { id = treaty_of_poznan_events.8 days = 1 popup = yes }
		}
	}

	#nfg annexes prussia
	option = {

		#remove the button var
		remove_variable = br_deciding_the_fate_of_prussia_button_var
