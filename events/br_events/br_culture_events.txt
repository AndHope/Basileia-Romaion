namespace = br_ere_culture_change
#Reminder to self use this style event for forming Indke and Aepoedian cultures.
br_ere_culture_change.1 = {
	type = country_event
	placement = c:br_kappadocia
	
	title = br_ere_culture_change.1.t
	desc = br_ere_culture_change.1.d
	flavor = br_ere_culture_change.1.f
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	duration = 3

	#orphan = yes #not yet in use
	
	trigger = {
		#decision
	}

	option = {
		name = br_ere_culture_change.1.a
		default_option = yes

		hidden_effect = {
		c:br_kappadocia ?= {
			#remove_primary_culture = cu:greek
			add_primary_culture = cu:cappadocian
			every_scope_pop = {
				change_pop_culture = { target = cu:cappadocian value = 0.55 }
			}
		}
		c:br_mesopotamia ?= {
			every_scope_pop = {
				change_pop_culture = { target = cu:cappadocian value = 0.2 }
			}
		}
		s:STATE_ANKARA ?= {
			add_homeland = cu:cappadocian
		}
		s:STATE_ADANA ?= {
			add_homeland = cu:cappadocian
		}
		s:STATE_DIYARBAKIR ?= {
			add_homeland = cu:cappadocian
		}

		custom_tooltip = {
			text = br_ere_culture_change_cappadocian
		}
	}
	}
}
