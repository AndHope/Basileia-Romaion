#Serbia
br_nation_startup_event.1 = {
	type = country_event
	placement = c:SER.capital

	title = br_nation_startup_event.1.t
	desc = br_nation_startup_event.1.d
	flavor = br_nation_startup_event.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/sepoy_mutiny"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	trigger = {   
	}
	immediate = {
		hidden_effect = {
			add_treasury = 250000
			add_loyalists = {
				pop_type = aristocrats
				value = medium_radicals
			}

				pop_type = officers
				value = very_large_radicals
			}
			add_loyalists = {
				pop_type = soldiers
				value = very_large_radicals
			}
			add_loyalists = {
				pop_type = clergymen
				value = very_large_radicals
			}
			add_loyalists = {
				pop_type = bureaucrats
				value = very_large_radicals
			}
			add_radicals = {
				pop_type = peasants
				value = medium_radicals
			}
		}
	}

	option = {
		name = br_nation_startup_event.1.a
		default_option = yes
		
		ai_chance = {
			base = 100
		}

	}

}
