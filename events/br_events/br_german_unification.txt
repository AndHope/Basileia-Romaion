# North/South German Confederation
br_german_unification.1 = {
	type = country_event
	placement = ROOT

	title = br_german_unification.1.t
	desc = {
		first_valid = {		
			triggered_desc = {
				desc = br_german_unification.1.d1
				trigger = {
					country_has_a_north_german_primary_culture = yes
				}
			}
			triggered_desc = {
				desc = br_german_unification.1.d2
				trigger = {
					country_has_a_south_german_primary_culture = yes
				}
			}
		}
	}
	flavor = br_german_unification.1.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		set_variable = je_german_confederate_unifier
	
		every_country = {
			limit = {
				country_rank < rank_value:great_power			
				OR = {
					AND = {
						root = { country_has_a_north_german_primary_culture = yes}
						country_has_a_north_german_primary_culture = yes
					}
					AND = {
						root = { country_has_a_south_german_primary_culture = yes }
						country_has_a_south_german_primary_culture = yes
					}					
				}
				NOT = {
					THIS = ROOT
				}				
			}
			
			if = {
				limit = {
					is_ai = yes			
					OR = {
						is_in_customs_union_with = root
						is_supporting_unification_candidate = { who = root country_formation = GER }
					}
				}			
				root = { annex_with_incorporation = prev }
			}
			# else = {
			# 	trigger_event = { id = br_german_unification.2 days = 0 popup = yes }
			# }
		}
	}

	option = {
		name = br_german_unification.1.a
		default_option = yes
		every_interest_group = {
			limit = {
				is_in_government = yes
			}
			leader = {
				add_modifier = {
					name = br_modifier_german_unification_popularity
					days = long_modifier_time
				}
			}
		}
		if = {
			limit = {
				country_has_a_north_german_primary_culture = yes
				NOT = { exists = c:NGF }
				OR = {
					country_tier = kingdom
					country_tier = principality
				}
			}
			
			# Reconcile with Austria if we didn't conquer states from them or release any crucial territories
			# hidden_effect = {
			# 	if = {
			# 		limit = {
			# 			exists = c:AUS
			# 			c:AUS = {
			# 				country_rank >= rank_value:great_power
			# 				NOT = { has_claim = root }
			# 				owns_entire_state_region = STATE_BOHEMIA
			# 				owns_entire_state_region = STATE_CENTRAL_HUNGARY
			# 				owns_entire_state_region = STATE_LOMBARDY
			# 			}
			# 		}
					
			# 		c:AUS = {
			# 			set_mutual_secret_goal = {
			# 				country = root
			# 				secret_goal = reconcile
			# 			}						
			# 		}
			# 	}
			# }		
			
			change_tag = NGF
		}
		else_if = {
			limit = {
				country_has_a_south_german_primary_culture = yes
				NOT = { exists = c:SGF }
				OR = {
					country_tier = kingdom
					country_tier = principality
				}
			}
			change_tag = SGF
		}
	}
	option = {
		name = br_german_unification.1.b
		add_modifier = {
			name = br_modifier_partial_german_unification_influence_prestige
			days = normal_modifier_time
		}
		if = {
			limit = {
				country_has_a_north_german_primary_culture = yes
				NOT = { exists = c:NGF }
				OR = {
					country_tier = kingdom
					country_tier = principality
				}
			}
			change_tag = NGF
		}
		else_if = {
			limit = {
				country_has_a_south_german_primary_culture = yes
				NOT = { exists = c:SGF }
				# OR = {
				# 	NOT = { exists = c:AUS }
				# 	NOT = { THIS = c:AUS }
				# }
				OR = {
					country_tier = kingdom
					country_tier = principality
				}
			}
			change_tag = SGF
		}
	}
}

# German National Identity
br_german_unification.5 = {
	type = country_event
	placement = ROOT

	title = br_german_unification.5.t
	flavor = br_german_unification.5.f
	desc = {
		first_valid = {
			triggered_desc = {
				desc = br_german_unification.5.d1
				trigger = {
					this = scope:german_unifier_candidate
				}
			}
			triggered_desc = {
				desc = br_german_unification.5.d2
				trigger = {
					exists = scope:german_unifier_candidate
				}
			}
			triggered_desc = {
				desc = br_german_unification.5.d3
				trigger = {
					always = yes
				}
			}			
		}
	}

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {}

	option = {
		name = br_german_unification.5.a
		default_option = yes	
		if = {
			limit = { 
				country_has_a_north_german_primary_culture = yes
			 }
			add_journal_entry = {
				type = je_north_german_unification
			}
		}
		else_if = {
			limit = { 
				country_has_a_south_german_primary_culture = yes
			 }
			add_journal_entry = {
				type = je_south_german_unification
			}
		}		
	}
}
